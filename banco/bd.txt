-- Criar banco de dados
CREATE DATABASE filmes_series;
USE filmes_series;

-- Criar tabela de filmes/séries
CREATE TABLE catalogo (
  id INT AUTO_INCREMENT PRIMARY KEY,
  titulo VARCHAR(100),
  genero VARCHAR(50),
  ano INT
);
